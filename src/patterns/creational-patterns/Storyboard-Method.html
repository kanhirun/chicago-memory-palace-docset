Creational Patterns

Storyboard Method

Intent

You have a UI class that you want to leverage Interface Builder with while still making it testable.

Code Example 

@IBDesignable
final class CustomLabel: UILabel {

    @IBInspectable var foo = 0.0

    // MARK: - Initializers

    required convenience init?(coder aDecoder: NSCoder) {
        self.init(aDecoder)
    }

    // This must be overridden to prevent the agent from crashing
    // in storyboards
    override init(frame: CGRect) {
        self.init(frame)
    }

    // Override this method to update the view while on IB
    override func prepareForInterfaceBuilder() {
        super.prepareForInterfaceBuilder()
    }

    // The designated initializer; Use this whenever you need to test too!
    init(_ aDecoder: NSCoder? = nil, _ frame: CGRect = CGRect.zero) {
        if let aDecoder = aDecoder {
            super.init(coder: aDecoder)
        } else {
            super.init(frame: frame)
        }
    }
}
