Test Console Pattern

Installing

```
npm install sfarthin/intercept-stdout
```

Code Example

```
const intercept = require('intercept-stdout');  // captures stdout

const printSumOfInversions = require("../src/printSumOfInversions.js").printSumOfInversions;

describe("printSumOfInversions", () => {
  it("prints Hello world!", () => {
    var output = "";
    const unhookIntercept = intercept( text => {
      output += text;
    });

    printSumOfInversions();

    unhookIntercept();
    expect(output).toEqual("Hello world!");
  })
});
```
